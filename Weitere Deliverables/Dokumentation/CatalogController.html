def approveUpload
    data = JSON.parse(request.body.read)
    catalog_year = data["year"].to_i
    if(catalog_year <= 1800)
        head :not_acceptable
    elsif(Catalog.exists?(year: catalog_year))
        head :bad_request
    else 
        head :ok
    end
end

def countChops
    catalog_year = params[:year].to_i
    if(Catalog.exists?(year: catalog_year))
        catalog = Catalog.find_by(year: catalog_year)
        @count = Chop.all.where(:catalog_id => catalog.id).size.to_i
        render :plain => @count
    else
        head 404
    end
end

def create 
    require 'csv'
    catalogYear = params[:year].to_i
    filename = params[:catalog].original_filename
    # only create new catalog when id doesnt exist already

    if (Catalog.exists?(:year => catalogYear))
        catalog_id = Catalog.find_by(:year => catalogYear).id
        Transition.where(catalog_id_new: catalog_id).or(Transition.where(catalog_id_old: catalog_id)).delete_all
        Chop.where(:catalog_id => catalog_id).delete_all
        Catalog.where(id: catalog_id).delete_all
    end

    catalog = Catalog.where(:year => catalogYear).first_or_create
    catalog.save
    
    # create an array with objects to bulk insert them into the DB
    bulk = []
    CSV.foreach(params[:catalog], liberal_parsing:true , col_sep: ";", quote_char:'"', headers: true) do |row|
        complement = "#{row["codable"]}"        
        if(complement == "oui" || complement == "complement")
            bulk.push({code: row["zcode"], description: row["text"], complement: (complement=="complement"), catalog_id: catalog.id})
        end
    end
    # If catalog didn't exist already, nothing will be deleted. Else all chops of this catalog are deleted and replaced with the new ones
    Chop.where(:catalog_id => catalog.id).delete_all
    result = Chop.insert_all(bulk)      # this is considerably faster than inserting each chop individually
    head :created
end
